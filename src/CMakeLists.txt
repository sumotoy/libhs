include(CheckSymbolExists)
check_symbol_exists(stpcpy string.h HAVE_STPCPY)
check_symbol_exists(asprintf stdio.h HAVE_ASPRINTF)
add_definitions(-DHAVE_CONFIG_H)
configure_file(config.h.in config.h)

set(HS_SOURCES ../include/hs.h
               ../include/hs/common.h
               ../include/hs/device.h
               ../include/hs/hid.h
               ../include/hs/monitor.h
               ../include/hs/platform.h
               ../include/hs/serial.h

               common.c
               compat.c
               device.c
               device_priv.h
               htable.c
               list.h
               monitor.c
               monitor_priv.h
               platform.c
               util.h)
if(WIN32)
    list(APPEND HS_SOURCES device_win32.c
                           device_win32_priv.h
                           hid_win32.c
                           monitor_win32.c
                           platform_win32.c
                           serial_win32.c)
else()
    list(APPEND HS_SOURCES device_posix.c
                           device_posix_priv.h
                           serial_posix.c)

    if(LINUX)
        list(APPEND HS_SOURCES hid_linux.c
                               monitor_linux.c
                               platform_posix.c)
    elseif(APPLE)
        list(APPEND HS_SOURCES hid_darwin.c
                               monitor_darwin.c
                               platform_darwin.c)
    else()
        message(FATAL_ERROR "Unsupported platform")
    endif()
endif()

add_library(hs ${HS_SOURCES})
target_link_libraries(hs ${HS_LINK_LIBRARIES})
